export $(grep -v '^#' ".env.local" | xargs)

#Update the inbound.json accrdong to your setup

#Update the dispatch-rule.json according to your setup
lk sip dispatch create ./trunking/plivo/inbound/dispatch-rule.json
#This will give a SIPDispatchRuleID, update it in .env
